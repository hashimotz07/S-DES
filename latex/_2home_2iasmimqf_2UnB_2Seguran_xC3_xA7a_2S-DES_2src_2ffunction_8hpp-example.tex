\begin{DoxyParams}{Parâmetros}
{\em n} & Bloco de 8 bits. \\
\hline
\end{DoxyParams}
\begin{DoxyReturn}{Retorna}
pair$<$int, int$>$ Par (L, R), onde L = 4 bits MSB, R = 4 bits LSB. 
\end{DoxyReturn}
\begin{DoxyPrecond}{Pré-\/condição}
n deve ser um valor de 8 bits. 
\end{DoxyPrecond}
\begin{DoxyPostcond}{Pós-\/condição}
